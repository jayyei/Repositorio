// Tipos de datos en Java

//Datos primitivos en los cuales resaltamos 4 categorias

	//Enteros : Referencias a numeros enteros. Se dividen en dos tipos
		int: 4 bytes de espacio para almacenamiento. Desde -2,147,483,648 hasta 2,147,483,647
		short: 2 bytes de espacio de almacenamiento. Desde -32,768 hasta 32,768
		long: 8 bytes de espacio para almacenamiento. Una barbaridad. Sufijo L 
		byte: 1 byte de espacio de alamacenamiento. Desde -128 hasta 127.	
	
	//Coma flotante o decimales, se dividen en dos categorias
		float: 4 bytes de espacio de almacenamiento. Aproximadamente 6 a 7 cifras decimales significativas. sufijo F.
		double: 8 bytes de espacio de almacenamiento. Aproximadamente 15 cifras significativas.

	//char o caracteres. Siempre iran entre comillas simples

	//Boolean: Toman dos valores o true o false 

//Variables : espacio en memoria donde se almacenara un valor que podra cambiar durante la ejecucion del programa. Para guardar datos temporales
	Ejemplo: int salario // Primero se especifica el tipo y despues el nombre de la variable.
